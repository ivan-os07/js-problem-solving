push 5, Кладёт число 5 на стек,stack = [5]
load x, Берёт значение переменной x и кладёт на стек,mem.x = 10 → stack.push(10)
store x,    Берёт верхушку стека и сохраняет в x,"stack = [7] → mem.x = 7, стек пуст"
add,    "Берёт два числа со стека, складывает, кладёт результат",3 4 add → 7
sub,    Вычитает: верхнее − нижнее,5 3 sub → 2
mul,    Умножение,4 5 mul → 20
mod,    Остаток от деления,10 3 mod → 1
eq, "Сравнение: 1 если равны, иначе 0",5 5 eq → 1
jmp L1, Безусловно перейти к метке L1,pc = labels['L1']
jz L1,  "Перейти к L1, только если верх стека = 0",0 jz L1 → переход
print,  Вывести верхушку стека в консоль,print → 120
read,   "Остановить, ждать ввода, положить число на стек"
halt,   Завершить программу,process.exit(0)
